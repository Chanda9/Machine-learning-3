#creating a function to replace all numeric values in mpg column as high, low or medium fuel efficiency 
def change_mpg_to_effi(cols):
    a=cols[0]
    if a<=17:
        return 'low'
    elif (a>17 and a<=29):
        return 'medium'
    else:
        return 'high'
